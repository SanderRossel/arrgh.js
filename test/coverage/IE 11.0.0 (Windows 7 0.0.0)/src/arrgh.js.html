    function stableQuicksort (map, startIndex, endIndex, compare) {
        var low = startIndex;
        var high = endIndex;
        var pindex = Math.floor((low + high) / 2);
        var pivot = map[pindex];
        var lindex;
        var hindex;
        var result;
 
        while (low <= high) {
            lindex = map[low];
            result = compare(lindex, pivot);
            while (result < 0 || (result === 0 && lindex < pivot)) {
                low += 1;
                lindex = map[low];
                result = compare(lindex, pivot);
            }
 
            hindex = map[high];
            result = compare(hindex, pivot);
            while (result > 0 || (result === 0 && hindex > pivot)) {
                high -= 1;
                hindex = map[high];
                result = compare(hindex, pivot);
            }
            if (low <= high) {
                var temp = map[low];
                map[low] = map[high];
                map[high] = temp;
                low += 1;
                high -= 1;
            }
        }
 
        if (low < endIndex) {
            stableQuicksort(map, low, endIndex, compare);
        }
        if (high > startIndex) {
            stableQuicksort(map, startIndex, high, compare);
        }
    }
